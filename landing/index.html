App.java 
import com.ligero.Ligero;

public class App {
    public static void main(String[] args) {
        Ligero app = Ligero.create(8080);
        app.get("/hello/{name}", ctx ->
            ctx.json(Map.of("message",
                "Hello, " + ctx.pathParam("name") + "!")));
        app.start();
    }
}
